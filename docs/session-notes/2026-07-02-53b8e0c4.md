function is_adjacent(tile1, tile2) {
    var dx = abs(tile1.grid_x - tile2.grid_x)
    var dy = abs(tile1.grid_y - tile2.grid_y)

function apply_gravity() {
    for (var xx = 0; xx < 6; xx++) {
        // Collect tiles from bottom to top
        var tiles = ds_list_create()
        for (var yy = 5; yy >= 0; yy--) {
            if (grid[# xx, yy] != noone) {
                ds_list_add(tiles, grid[# xx, yy])
            }
        }
        
        // Clear column
        for (var yy = 0; yy < 6; yy++) {
            grid[# xx, yy] = noone
        }
        
        // Fill from bottom with collected tiles
        for (var i = 0; i < ds_list_size(tiles); i++) {
            var yy = 5 - i
            var tile = tiles[| i]
            grid[# xx, yy] = tile
            tile.grid_x = xx
            tile.grid_y = yy
            tile.target_x = 100 + xx * 32
            tile.target_y = 100 + yy * 32
        }
        
        ds_list_destroy(tiles)
    }
}
